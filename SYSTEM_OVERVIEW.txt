================================================================================
MODEL ARCHITECTURE
================================================================================

Input (24 hours × features)
    ↓
Conv1D (64 filters) + ReLU + MaxPool + Dropout
    ↓
Conv1D (32 filters) + ReLU + MaxPool + Dropout
    ↓
LSTM (50 units, return_sequences=True) + Dropout
    ↓
LSTM (50 units) + Dropout
    ↓
Dense (25 units) + ReLU + Dropout
    ↓
Dense (1 unit) - Output

Total Parameters: ~50,000
Training Time: 15 min (CPU) / 3 min (GPU)
Inference: <100ms

USAGE EXAMPLES
================================================================================

Python API Usage:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    import requests

    # Get 6-hour forecast
    response = requests.post('http://localhost:5000/predict',
                            json={'hours_ahead': 6})
    forecast = response.json()

    for hour in forecast['forecast']:
        print(f"Hour {hour['hour']}: {hour['predicted_traffic']:.1f} vehicles")
        print(f"  Congestion: {hour['congestion_level']}")

cURL Usage:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    # Get prediction
    curl -X POST http://localhost:5000/predict \
      -H "Content-Type: application/json" \
      -d '{"hours_ahead": 6}'

    # Get current status
    curl http://localhost:5000/current

Direct Model Usage:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
    from model import TrafficPredictionModel

    model = TrafficPredictionModel()
    model.load_model('models/traffic_model_v20260309.h5')

    predictions = model.predict_future(recent_data, hours_ahead=12)

Example 1: Get Traffic Prediction
────────────────────────────────────────────────────────────────────────────────
    import requests

    response = requests.post('http://localhost:5000/predict',
                            json={'hours_ahead': 6})
    forecast = response.json()

    for hour in forecast['forecast']:
        print(f"Hour {hour['hour']}: {hour['predicted_traffic']:.1f} vehicles")


Example 2: Use Model Directly
────────────────────────────────────────────────────────────────────────────────
    from model import TrafficPredictionModel

    model = TrafficPredictionModel()
    model.load_model('models/traffic_model_v20260309.h5')
    predictions = model.predict_future(recent_data, hours_ahead=12)


Example 3: Generate Custom Data
────────────────────────────────────────────────────────────────────────────────
    from data_generator import TrafficDataGenerator
    from datetime import datetime

    generator = TrafficDataGenerator()
    df = generator.generate_traffic_data(
        start_date=datetime(2023, 1, 1),
        n_days=365,
        save_to_db=True
    )
